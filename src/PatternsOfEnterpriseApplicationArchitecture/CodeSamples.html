Transaction Script Example - C#

public class OrderReportTransactionScript
{
    public void PrintOrdersWithTotals()
    {
        var ordersDataSet = GetAllOrders();

        PrintReportToConsole(ordersDataSet);
    }

    private DataSet GetAllOrders()
    {
        var connectionString = "[My Connection String]";

        var allOrdersSql = @"
SELECT
    *
FROM
    Order

SELECT
    *
FROM
    LineItem";

        var ordersDataSet = new DataSet();

        using (var connection = new SqlConnection(connectionString))
        {
            var command = new SqlCommand(allOrdersSql, connection);

            var sqlDataAdapter = new SqlDataAdapter(command);

            connection.Open();
            sqlDataAdapter.Fill(ordersDataSet);
            connection.Close();
        }

        return ordersDataSet;
    }

    private void PrintReportToConsole(DataSet ordersDataSet)
    {
        Console.WriteLine("Date\tName\tTotal");

        foreach (DataRow order in ordersDataSet.Tables[0].Rows)
        {
            var orderTotal = CalculateOrderTotal(order, ordersDataSet.Tables[1]);
            var orderDate = Convert.ToDateTime(order["OrderDate"]).ToShortDateString();
            var customerName = order["CustomerName"].ToString();

            Console.WriteLine(
                "'{0}'\t'{1}'\t{2}",
                orderDate,
                customerName,
                orderTotal);
        }
    }

    private decimal CalculateOrderTotal(DataRow order, DataTable lineItems)
    {
        var orderTotal = 0m;

        var lineItemsForOrder = lineItems.Select("OrderId = " + order["OrderId"]);

        foreach (DataRow lineItem in lineItemsForOrder)
        {
            var unitPrice = Convert.ToDecimal(lineItem["UnitPrice"]);
            var quanity = Convert.ToInt32(lineItem["Quantity"]);

            orderTotal += unitPrice * quanity;
        }

        return orderTotal;
    }
}

Domain Model Example

public class OrderForDisplay
{
    private readonly string _customerFirstName;
    private readonly string _customerLastName;

    public OrderForDisplay(
        Guid orderId,
        string customerFirstName,
        string customerLastName,
        DateTime orderDate)
    {
        _customerLastName = customerLastName;
        _customerFirstName = customerFirstName;
        OrderId = orderId;
        OrderDate = orderDate;
    }

    public Guid OrderId { get; private set; }
    public DateTime OrderDate { get; private set; }

    public string GetFullCustomerName()
    {
        return string.Format("{0} {1}", _customerFirstName, _customerLastName);
    }
}

Table Module Example

public class PortfolioTableModule
{
    private readonly DataSet _dataSet;

    public PortfolioTableModule(
        DataSet dataSet)
    {
        _dataSet = dataSet;
    }

    public DataRow this[int number]
    {
        get
        {
            var filter = string.Format("Number = {0}", number);

            return _dataSet.Tables["Portfolio"].Select(filter)[0];
        }
    }

    public int[] GetBenchmarkNumbers(
        int portfolioNumber)
    {
        var filter = string.Format("Portfolio = {0}", portfolioNumber);

        var dataRows = _dataSet.Tables["Benchmark"].Select(filter);

        return dataRows.Select(r => (int)r["Number"]).ToArray();
    }

    public string GetDisplayName(
        int portfolioNumber)
    {
        var dataRow = this[portfolioNumber];

        return string.Format("{0} - {1}", dataRow["Number"], dataRow["Name"]);
    }
}
