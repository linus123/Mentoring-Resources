Design Patterns

Decorator Pattern

public interface IShape
{
    string GetShapeDescription();
}

public class Circle : IShape
{
    private decimal _radius;

    public Circle(decimal radius)
    {
        _radius = radius;
    }

    public string GetShapeDescription()
    {
        return string.Format("A circle of radius {0}", _radius);
    }
}

public class Square : IShape
{
    private decimal _widthHeight;

    public Square(decimal widthHeight)
    {
        _widthHeight = widthHeight;
    }

    public string GetShapeDescription()
    {
        return string.Format("A square of with width/height {0}", _widthHeight);
    }
}

public class ColoredShape : IShape
{
    private IShape _shape;
    private string _color;

    public ColoredShape(IShape shape, string color)
    {
        _color = color;
        _shape = shape;
    }

    public string GetShapeDescription()
    {
        return string.Format("{0} which is colored {1}.", _shape.GetShapeDescription(), _color);
    }
}

public class Program
{
    public void Main()
    {
        var circle = new Circle(1.2m);
        var redCircle = new ColoredShape(circle, "Red");
        Console.WriteLine(redCircle.GetShapeDescription());
        // A circle of radius 1.2 which is colored Red

        var square = new Square(3.5m);
        var blueSquare = new ColoredShape(square, "Blue");
        Console.WriteLine(blueSquare.GetShapeDescription());
        // A square of with width/height 3.5 which is colored Blue
    }
}
