Customer Repository

public class CustomerRepository
{
    private ICustomerDataGateway _customerDataGateway;

    public CustomerRepository(
        ICustomerDataGateway customerDataGateway)
    {
        _customerDataGateway = customerDataGateway;
    }

    public Customer Save(
        Customer customer)
    {
        if (customer.IsNew)
        {
            var customerDto = new CustomerDto(){
                FirstName = customer.FirstName,
                LastName = customer.LastName
            };

            var customerId = _customerDataGateway.Insert(CustomerDto dto);

            customer.Id = customerId;
        }
        else
        {
            var customerDto = new CustomerDto(){
                Id = customer.Id,
                FirstName = customer.FirstName,
                LastName = customer.LastName
            };

            _customerDataGateway.Save(CustomerDto dto)
        }

        return customer;
    }

    public Customer Get(
        int customerId)
    {
        var customerDto = _customerDataGateway.Get(customerId);

        return new Customer(){
            Id = customerDto.Id,
            FirstName = customerDto.FirstName,
            LastName = customerDto.LastName
        };
    }
}

public class Customer
{
    public int? Id { get; set; }
    public string FirstName { get; set; }
    public string LastName { get; set; }

    public bool IsNew
    {
        get { return !Id.HasValue; }
    }
}

// Data Access

public interface ICustomerDataGateway
{
    int Insert(CustomerDto dto);
    void Update(CustomerDto dto);
    CustomerDto Get(int customerId);
}

public class CustomerDto
{
    public int Id { get; private set; }
    public string FirstName { get; set; }
    public string LastName { get; set; }
}
